version: "3"

services:

  hadoop-namenode:
    image: bde2020/hadoop-namenode:2.0.0-hadoop3.2.1-java8
    container_name: namenode
    environment:
      - CLUSTER_NAME=datalake-hadoop
      - CORE_CONF_fs_defaultFS=hdfs://namenode:9000
    ports:
      - "9870:9870"   # Web UI du NameNode
    volumes:
      - hadoop_namenode:/hadoop/dfs/name
    networks:
      - hadoop-net

  hadoop-datanode:
    image: bde2020/hadoop-datanode:2.0.0-hadoop3.2.1-java8
    container_name: datanode
    depends_on:
      - hadoop-namenode
    environment:
      - CORE_CONF_fs_defaultFS=hdfs://namenode:9000
    ports:
      - "9864:9864"   # Web UI du DataNode
    volumes:
      - hadoop_datanode:/hadoop/dfs/data
    networks:
      - hadoop-net

  spark:
    image: bitnami/spark:3.3.1
    container_name: spark
    environment:
      - SPARK_MODE=master
      - SPARK_MASTER_HOST=spark
      - SPARK_MASTER_PORT=7077
    ports:
      - "8080:8080"  # Web UI Spark Master
      - "4040:4040"  # Spark Application UI
    networks:
      - hadoop-net
    volumes:
      - ./spark-app:/app  # Ton code PySpark ici
    depends_on:
      - hadoop-namenode
      - hadoop-datanode

volumes:
  hadoop_namenode:
  hadoop_datanode:

networks:
  hadoop-net:
    driver: bridge
